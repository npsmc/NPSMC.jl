# Catalog

using Plots, NPSMC

σ = 10.0 ρ = 28.0 β = 8.0/3

dtintegration = 0.01 dtstates = 1 dtobs = 8 parameters = [σ, ρ, β] varobs = [1] nblooptrain = 100 nblooptest = 10 sigma2catalog = 0.0 sigma2obs = 2.0

ssm = StateSpaceModel( lorenz63, dtintegration, dtstates, dtobs, parameters, varobs, nblooptrain, nblooptest, sigma2catalog, sigma2obs )

xt, yo, catalog = generate_data( ssm , [10.0;0.0;0.0]);

plot(catalog.analogs[1,:]) plot!(catalog.analogs[2,:]) plot!(catalog.analogs[3,:])

p = plot3d(1, xlim=(-25,25), ylim=(-25,25), zlim=(0,50), title = "Lorenz 63", marker = 1) for x in eachcol(catalog.analogs) push!(p, x...) end p
