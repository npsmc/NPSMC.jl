using Plots, NPSMC
α = 3.0

dt_integration = 0.01
dt_states      = 1 
dt_obs         = 8 
params         = [α]
var_obs        = [1]
nb_loop_train  = 10^2 
nb_loop_test   = 10
sigma2_catalog = 0.0
sigma2_obs     = 0.1

ssm = StateSpaceModel( sinus, 
                       dt_integration, 
                       dt_states, 
                       dt_obs, 
                       params, 
                       var_obs,
                       nb_loop_train, 
                       nb_loop_test,
                       sigma2_catalog, 
                       sigma2_obs )

xt, yo, catalog = generate_data( ssm, [0.0] );

plot(xt.time, vcat(xt.values...)[:,1])
scatter!(yo.time, vcat(yo.values...)[:,1]; markersize=2)
scatter(catalog.analogs[1,:], catalog.successors[1,:], markersize=1)
